-- ========== FUNGSI LOG ==========
local function addLog(msg, level)
    level = level or "info"
    local color
    if     level == "err"   then color = Color3.fromRGB(255, 80, 80)
    elseif level == "warn"  then color = Color3.fromRGB(255,200, 60)
    elseif level == "ok"    then color = Color3.fromRGB(60, 220,100)
    elseif level == "debug" then color = Color3.fromRGB(130,130,170)
    else                         color = Color3.fromRGB(180,210,255) end

    local elapsed = tick() - logStartTick
    local ts = string.format("[%02d:%02d:%02d]",
        math.floor(elapsed/3600), math.floor(elapsed/60)%60, math.floor(elapsed)%60)
    local fullText = ts .. " " .. tostring(msg)

    logOrderCount = logOrderCount + 1

    -- Buat label baru di bagian atas (order terkecil = teratas)
    local lbl = Instance.new("TextLabel")
    lbl.Size       = UDim2.new(1,0,0,0)
    lbl.AutomaticSize = Enum.AutomaticSize.Y
    lbl.BackgroundTransparency = 1
    lbl.TextColor3 = color
    lbl.Font       = Enum.Font.Code
    lbl.TextSize   = 11
    lbl.TextWrapped= true
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.TextYAlignment = Enum.TextYAlignment.Top
    lbl.Text       = fullText
    lbl.LayoutOrder= -logOrderCount   -- negatif agar terbaru paling atas
    lbl.ZIndex     = 10012
    lbl.Parent     = logBody

    table.insert(logEntries, 1, { text=fullText, color=color })
    table.insert(logLabels,  1, lbl)

    -- Hapus entri terlama jika melebihi limit
    if #logEntries > LOG_MAX then
        local old = table.remove(logLabels)
        if old then old:Destroy() end
        table.remove(logEntries)
    end

    -- Update count di judul
    logTitle.Text = string.format("📋 Debug Log (%d)", #logEntries)

    -- Scroll ke atas untuk lihat log terbaru
    logBody.CanvasPosition = Vector2.new(0,0)
end

-- ========== OVERRIDE PRINT → CAPTURE KE LOG ==========
-- Harus didefinisikan SEBELUM notify agar _origPrint tersedia
local _origPrint = print
print = function(...)
    local parts = {}
    for i = 1, select("#", ...) do parts[i] = tostring(select(i,...)) end
    local msg = table.concat(parts, "\t")
    _origPrint(msg)
    -- Tentukan level dari prefix
    local lvl = "debug"
    if     msg:find("%[VERIF%].*TIDAK") or msg:find("NIL ⚠") then lvl = "err"
    elseif msg:find("%[VERIF%].*ditemukan")                   then lvl = "ok"
    elseif msg:find("%[EQUIP%].*NIL")                         then lvl = "warn"
    elseif msg:find("%[PENDING%].*masih")                     then lvl = "warn"
    elseif msg:find("%[EQUIP%]") or msg:find("%[PENDING%]") or msg:find("%[VERIF%]") then lvl = "info"
    end
    addLog(msg, lvl)
end

local function notify(msg, color)
    -- Tentukan level log berdasarkan warna atau prefix pesan
    local level = "info"
    if color then
        local r,g,b = color.R, color.G, color.B
        if r > 0.8 and g < 0.5 then level = "err"
        elseif r > 0.8 and g > 0.6 then level = "warn"
        elseif g > 0.7 and r < 0.5 then level = "ok"
        elseif b > 0.8 and r < 0.6 then level = "info" end
    end
    if msg:find("^✓") or msg:find("^✅") then level = "ok"
    elseif msg:find("^⚠") or msg:find("^⏳") then level = "warn"
    elseif msg:find("^✗") or msg:find("gagal") or msg:find("TIDAK") then level = "err" end

    addLog(msg, level)
    _origPrint(msg)   -- _origPrint: tidak trigger print-override, no double-log
    pcall(function()
        game:GetService("StarterGui"):SetCore("ChatMakeSystemMessage", {
            Text = msg, Color = color or Color3.fromRGB(0,220,80),
            FontSize = Enum.FontSize.Size18,
        })
    end)
end

local function getRawCoins()
    local ok, txt = pcall(function()
        return player.PlayerGui.Events.Frame.CurrencyCounter.Counter.Text
    end)
    if not ok or not txt then return 0 end
    txt = txt:gsub(",",""):gsub("%s","")
    local num = tonumber(txt)
    if num then return num end
    local val2, suf = txt:match("^([%d%.]+)([KMB]?)$")
    val2 = tonumber(val2) or 0
    if suf == "K" then return val2 * 1000
    elseif suf == "M" then return val2 * 1000000
    elseif suf == "B" then return val2 * 1000000000 end
    return val2
end

local function formatCoins(n)
    if n >= 1000000 then return string.format("$%.2fM", n/1000000)
    elseif n >= 1000 then return string.format("$%.2fK", n/1000)
    else return "$"..tostring(n) end
end

local function formatTime(seconds)
    local h = math.floor(seconds/3600)
    local m = math.floor((seconds%3600)/60)
    local s = math.floor(seconds%60)
    return string.format("%02dh %02dm %02ds", h, m, s)
end

local function formatRow1(caught, miss, fpm, ms)
    local msStr = ms > 0
        and string.format(" | <font color='#aaffaa'>~%dms</font>", math.floor(ms))
        or ""
    local delayStr = optimalDelay > 0
        and string.format(" | <font color='#ffcc66'>D:%.1f</font>", optimalDelay)
        or ""
    return string.format(
        "%d/<font color='#ff4444'>%d</font> | %.1f F/M%s%s",
        caught, miss, fpm, msStr, delayStr)
end

local function buildNameDB()
    local rs = game:GetService("ReplicatedStorage")
    local MODULE_FOLDERS = { "Baits", "Fishing Rods", "Rods" }

    row6Val.Text = "⏳ Preload module..."
    addLog("🔧 buildNameDB: mulai preload modul...", "debug")
    local loadedCount = 0
    for _, fn in ipairs(MODULE_FOLDERS) do
        local f = rs:FindFirstChild(fn)
        if f then
            addLog("🔧 buildNameDB: folder '" .. fn .. "' ditemukan (" .. #f:GetChildren() .. " anak)", "debug")
            for _, m in ipairs(f:GetChildren()) do
                if m:IsA("ModuleScript") then
                    local ok, err = pcall(require, m)
                    if ok then loadedCount = loadedCount + 1
                    else addLog("⚠ require gagal: " .. m.Name .. " → " .. tostring(err), "warn") end
                end
            end
        else
            addLog("⚠ buildNameDB: folder '" .. fn .. "' TIDAK ditemukan di ReplicatedStorage", "warn")
        end
    end
    addLog("🔧 buildNameDB: " .. loadedCount .. " modul di-require", "debug")
    task.wait(1)

    local gc = getgc(true)
    local rodDB     = {}
    local baitDB    = {}
    local rodTiers  = {}
    local baitTiers = {}
    local gcRodCount = 0; local gcBaitCount = 0
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local _, typ  = pcall(rawget, v, "Type")
            local _, id   = pcall(rawget, v, "Id")
            local _, name = pcall(rawget, v, "Name")
            local _, tier = pcall(rawget, v, "Tier")
            if type(id)=="number" and type(name)=="string" and name~="" and type(typ)=="string" then
                if typ == "Fishing Rods" then
                    if not rodDB[name] then rodDB[name] = id; gcRodCount = gcRodCount + 1 end
                    if type(tier)=="number" and not rodTiers[id] then
                        rodTiers[id] = { tier=tier, name=name }
                    end
                elseif typ == "Baits" then
                    if not baitDB[name] then baitDB[name] = id; gcBaitCount = gcBaitCount + 1 end
                    if type(tier)=="number" and not baitTiers[id] then
                        baitTiers[id] = { tier=tier, name=name }
                    end
                end
            end
        end
    end

    addLog(string.format("🔧 buildNameDB selesai: %d rod, %d bait ditemukan di GC", gcRodCount, gcBaitCount), "info")

    -- Debug: cek apakah target rod/bait ditemukan
    for _, rodName in ipairs(TARGET_RODS) do
        local id = rodDB[rodName]
        if id then
            local td = rodTiers[id]
            addLog(string.format("✅ ROD ditemukan: '%s' id=%d tier=%s", rodName, id, td and tostring(td.tier) or "NIL"), "ok")
        else
            addLog(string.format("❌ ROD tidak ditemukan di GC: '%s' — modul belum ter-load atau nama beda?", rodName), "err")
        end
    end
    for _, b in ipairs(TARGET_BAITS) do
        local id = baitDB[b.name]
        if id then
            local td = baitTiers[id]
            addLog(string.format("✅ BAIT ditemukan: '%s' id=%d tier=%s", b.name, id, td and tostring(td.tier) or "NIL"), "ok")
        else
            addLog(string.format("❌ BAIT tidak ditemukan di GC: '%s'", b.name), "err")
        end
    end

    return rodDB, baitDB, rodTiers, baitTiers
end

local function scanPriceDB(rodDB, baitDB, rodPriceDB, baitPriceDB, rodTiers, baitTiers)
    local rs = game:GetService("ReplicatedStorage")
    local MODULE_FOLDERS = { "Baits", "Fishing Rods", "Rods" }

    addLog("🔍 scanPriceDB: mulai scan harga...", "debug")

    local idToRodName  = {}; for name, id in pairs(rodDB)  do idToRodName[id]  = name end
    local idToBaitName = {}; for name, id in pairs(baitDB) do idToBaitName[id] = name end

    local needed = 0
    for _, rodName in ipairs(TARGET_RODS) do
        local id = rodDB[rodName]
        if id and not rodPriceDB[id] then needed = needed + 1 end
    end
    for _, baitEntry in ipairs(TARGET_BAITS) do
        local id = baitDB[baitEntry.name]
        if id and not baitPriceDB[id] then needed = needed + 1 end
    end
    addLog(string.format("🔍 scanPriceDB: butuh harga untuk %d item dari GC scan", needed), "debug")

    local targetRodIds  = {}
    local targetBaitIds = {}
    for _, rodName in ipairs(TARGET_RODS) do
        local id = rodDB[rodName]; if id then targetRodIds[id] = true end
    end
    for _, baitEntry in ipairs(TARGET_BAITS) do
        local id = baitDB[baitEntry.name]; if id then targetBaitIds[id] = true end
    end

    -- ── SCAN GC PASS 1 ──────────────────────────────────────────
    -- Hanya tabel tepat 2 field {Id, Price} = price table murni.
    -- Tabel field lebih (Totem/Shop) punya harga berbeda → skip.
    local gc = getgc(true)
    local gcPriceFound = 0
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local _, id    = pcall(rawget, v, "Id")
            local _, price = pcall(rawget, v, "Price")
            if type(id)=="number" and type(price)=="number" and price > 0 then
                local n = 0
                pcall(function() for _ in pairs(v) do n = n + 1 end end)
                if n == 2 then
                    if idToRodName[id] and not rodPriceDB[id] then
                        rodPriceDB[id] = price
                        addLog(string.format("✅ [GC] Harga rod '%s' id=%d: %s", idToRodName[id], id, formatCoins(price)), "ok")
                        gcPriceFound = gcPriceFound + 1
                        if targetRodIds[id] then needed = needed - 1 end
                    end
                    if idToBaitName[id] and not baitPriceDB[id] then
                        baitPriceDB[id] = price
                        addLog(string.format("✅ [GC] Harga bait '%s' id=%d: %s", idToBaitName[id], id, formatCoins(price)), "ok")
                        gcPriceFound = gcPriceFound + 1
                        if targetBaitIds[id] then needed = needed - 1 end
                    end
                end
            end
        end
        if needed <= 0 then break end
    end
    addLog(string.format("🔍 scanPriceDB GC pass: %d harga ditemukan, %d masih kurang", gcPriceFound, math.max(needed,0)), "debug")

    -- ── SCAN MODULE PASS 2 ──────────────────────────────────────
    local modPriceFound = 0
    for _, fn in ipairs(MODULE_FOLDERS) do
        local f = rs:FindFirstChild(fn)
        if f then for _, m in ipairs(f:GetChildren()) do
            if m:IsA("ModuleScript") then
                local ok, data = pcall(require, m)
                if ok and type(data)=="table" then
                    local name  = rawget(data, "Name") or m.Name
                    local price = rawget(data, "Price")
                    local tier  = rawget(data, "Tier")
                    if type(name)=="string" and name~="" and type(price)=="number" and price>0 then
                        local rid = rodDB[name]
                        local bid = baitDB[name]
                        if rid then
                            if not rodPriceDB[rid] then
                                addLog(string.format("✅ [MOD] Harga rod '%s' id=%d: %s (dari modul)", name, rid, formatCoins(price)), "ok")
                                modPriceFound = modPriceFound + 1
                            end
                            rodPriceDB[rid] = price
                            if type(tier)=="number" and not rodTiers[rid] then
                                rodTiers[rid] = { tier=tier, name=name }
                            end
                        end
                        if bid then
                            if not baitPriceDB[bid] then
                                addLog(string.format("✅ [MOD] Harga bait '%s' id=%d: %s (dari modul)", name, bid, formatCoins(price)), "ok")
                                modPriceFound = modPriceFound + 1
                            end
                            baitPriceDB[bid] = price
                            if type(tier)=="number" and not baitTiers[bid] then
                                baitTiers[bid] = { tier=tier, name=name }
                            end
                        end
                        -- Log modul yang ID-nya tidak dikenal (berguna untuk diagnosa nama beda)
                        if not rid and not bid then
                            addLog(string.format("🔎 [MOD] '%s' id? tidak cocok di rodDB/baitDB (tier=%s price=%s)", name, tostring(tier), formatCoins(price)), "debug")
                        end
                    end
                else
                    if not ok then
                        addLog(string.format("⚠ [MOD] require gagal: %s/%s", fn, m.Name), "warn")
                    end
                end
            end
        end end
    end
    addLog(string.format("🔍 scanPriceDB modul pass: %d harga baru dari modul", modPriceFound), "debug")

    -- ── HASIL AKHIR ─────────────────────────────────────────────
    local allFound = true
    for _, rodName in ipairs(TARGET_RODS) do
        local id = rodDB[rodName]
        if id then
            if rodPriceDB[id] then
                addLog(string.format("✅ Harga final rod '%s' id=%d: %s", rodName, id, formatCoins(rodPriceDB[id])), "ok")
                _nilPriceOnce["rod:" .. id] = nil
            else
                local k = "rod:" .. id
                if not _nilPriceOnce[k] then
                    _nilPriceOnce[k] = true
                    addLog(string.format("❌ Harga rod '%s' id=%d MASIH NIL — tidak ada di GC maupun modul!", rodName, id), "err")
                end
                allFound = false
            end
        else
            addLog(string.format("❌ Rod '%s' tidak punya ID di rodDB, skip harga", rodName), "err")
            allFound = false
        end
    end
    if allFound then
        for _, baitEntry in ipairs(TARGET_BAITS) do
            local id = baitDB[baitEntry.name]
            if id and baitPriceDB[id] then
                _nilPriceOnce["bait:" .. id] = nil
            elseif id and not baitPriceDB[id] then
                allFound = false
                local k = "bait:" .. id
                if not _nilPriceOnce[k] then
                    _nilPriceOnce[k] = true
                    addLog(string.format("❌ Harga bait '%s' id=%d NIL", baitEntry.name, id), "err")
                end
            end
        end
    end
    return allFound
end

local _gcCache = nil
local _ownedRodIdSetCache = nil

local function refreshGCCache()
    _gcCache = getgc(true)
    _ownedRodIdSetCache = nil
end

local function getOwnedRodsFromCache()
    local gc = _gcCache or getgc(true)
    local owned = {}; local seen = {}
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local okU, uuid = pcall(rawget, v, "UUID")
            local okI, id   = pcall(rawget, v, "Id")
            if okU and okI
                and type(uuid) == "string" and uuid:match(UUID_PAT)
                and type(id)   == "number"
                and not seen[uuid]
            then
                if rodTiers and rodTiers[id] then
                    seen[uuid] = true
                    owned[#owned+1] = { id = id, uuid = uuid }
                end
            end
        end
    end
    return owned
end

local function getOwnedRodIdSetFromCache()
    if _ownedRodIdSetCache then return _ownedRodIdSetCache end
    local gc = _gcCache or getgc(true)
    local ownedSet = {}
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local okE, eqBait = pcall(rawget, v, "EquippedBaitId")
            local okI, inv    = pcall(rawget, v, "Inventory")
            if okE and okI and type(eqBait)=="number" and type(inv)=="table" then
                local candidates = { "Rods", "Fishing Rods", "FishingRods", "Fishing_Rods" }
                for ci = 1, #candidates do
                    local key = candidates[ci]
                    local okR, rods = pcall(rawget, inv, key)
                    if okR and type(rods) == "table" then
                        pcall(function()
                            for _, rod in pairs(rods) do
                                if type(rod) == "table" then
                                    local okId, id = pcall(rawget, rod, "Id")
                                    if okId and type(id)=="number" then ownedSet[id] = true end
                                elseif type(rod) == "number" then
                                    ownedSet[rod] = true
                                end
                            end
                        end)
                    end
                end
                break
            end
        end
    end
    _ownedRodIdSetCache = ownedSet
    return ownedSet
end

local function getOwnedBaitsFromCache()
    local gc = _gcCache or getgc(true)
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local okE, eqBait = pcall(rawget, v, "EquippedBaitId")
            local okI, inv    = pcall(rawget, v, "Inventory")
            if okE and okI and type(eqBait)=="number" and type(inv)=="table" then
                local okB, baits = pcall(rawget, inv, "Baits")
                if okB and type(baits)=="table" then
                    local owned = {}
                    pcall(function()
                        for _, bait in pairs(baits) do
                            if type(bait)=="table" then
                                local okId, id = pcall(rawget, bait, "Id")
                                if okId and type(id)=="number" then
                                    owned[#owned+1] = { id=id }
                                end
                            end
                        end
                    end)
                    return owned
                end
            end
        end
    end
    return {}
end

local function getOwnedRods()  return getOwnedRodsFromCache()  end
local function getOwnedBaits() return getOwnedBaitsFromCache() end

local function isRodOwned(targetId, cachedRods)
    local rods = cachedRods or getOwnedRods()
    for _, rod in ipairs(rods) do if rod.id == targetId then return true end end
    local idSet = getOwnedRodIdSetFromCache()
    return idSet[targetId] == true
end

local function isBaitOwned(targetId, cachedBaits)
    local baits = cachedBaits or getOwnedBaits()
    for _, bait in ipairs(baits) do if bait.id == targetId then return true end end
    return false
end

local function getUUIDs()
    local gc = _gcCache or getgc(true); local t = {}
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local ok, uuid = pcall(rawget, v, "UUID")
            if ok and type(uuid)=="string" and uuid:match(UUID_PAT) then t[uuid]=true end
        end
    end
    return t
end

local function getNewUUIDs(before, timeout)
    local deadline = tick() + (timeout or 3)
    while tick() < deadline do
        local gc = getgc(true)
        local new = {}
        for i = 1, #gc do
            local v = gc[i]
            if type(v) == "table" then
                local ok, uuid = pcall(rawget, v, "UUID")
                if ok and type(uuid)=="string" and uuid:match(UUID_PAT) and not before[uuid] then
                    new[#new+1] = uuid
                end
            end
        end
        if #new > 0 then return new end
        task.wait(0.1)
    end
    return {}
end

local function getNextObjectiveForQuest(qname)
    local node = findQuestNodeByName(qname)
    if not node then return nil end
    local content = node:FindFirstChild("Content")
    if not content then return nil end
    local objectives = {}
    for _, child in ipairs(content:GetChildren()) do
        if child.Name:match("^Objective%d+$") then objectives[#objectives+1] = child end
    end
    table.sort(objectives, function(a,b)
        return (tonumber(a.Name:match("%d+")) or 0) < (tonumber(b.Name:match("%d+")) or 0)
    end)
    for _, obj in ipairs(objectives) do
        local okProg, progTxt = pcall(function() return obj.BarFrame.Progress.Text end)
        if not okProg or not progTxt then continue end
        local clean = progTxt:gsub(",","")
        local cur, max = clean:match("(%d+)%s*/%s*(%d+)")
        cur = tonumber(cur) or 0; max = tonumber(max) or 1
        if cur < max then
            local label = obj.Name
            local okLbl, lbl = pcall(function()
                local display = obj.Content.Display
                local parts = {}
                for _, c in ipairs(display:GetChildren()) do
                    if c:IsA("TextLabel") then
                        local okT, t = pcall(function() return c.Text end)
                        if okT and type(t) == "string" and t ~= "" and t ~= "Purchase a" then
                            parts[#parts+1] = { order = c.LayoutOrder or 0, name = c.Name, text = t }
                        end
                    end
                end
                table.sort(parts, function(a, b)
                    if a.order ~= b.order then return a.order < b.order end
                    return a.name < b.name
                end)
                local out = {}
                for i = 1, #parts do out[i] = parts[i].text end
                return table.concat(out, " "):gsub("%s+", " "):gsub("^%s*(.-)%s*$","%1")
            end)
            if okLbl and lbl and lbl ~= "" then label = lbl end

        end
    end
    return nil
end

local function getNextObjective()
    local node = findDeepSeaNode()
    if not node then return nil end
    local content = node:FindFirstChild("Content")
    if not content then return nil end
    local objectives = {}
    for _, child in ipairs(content:GetChildren()) do
        if child.Name:match("^Objective%d+$") then objectives[#objectives+1] = child end
    end
    table.sort(objectives, function(a,b)
        return (tonumber(a.Name:match("%d+")) or 0) < (tonumber(b.Name:match("%d+")) or 0)
    end)
    for _, obj in ipairs(objectives) do
        local okProg, progTxt = pcall(function() return obj.BarFrame.Progress.Text end)
        if not okProg or not progTxt then continue end
        local clean = progTxt:gsub(",","")
        local cur, max = clean:match("(%d+)%s*/%s*(%d+)")
        cur = tonumber(cur) or 0; max = tonumber(max) or 1
        if cur < max then
            local label = obj.Name
            local okLbl, lbl = pcall(function()
                local display = obj.Content.Display
                local parts = {}
                for _, c in ipairs(display:GetChildren()) do
                    if c:IsA("TextLabel") then
                        local okT, t = pcall(function() return c.Text end)
                        if okT and type(t) == "string" and t ~= "" and t ~= "Purchase a" then
                            parts[#parts+1] = { order = c.LayoutOrder or 0, name = c.Name, text = t }
                        end
                    end
                end
                table.sort(parts, function(a, b)
                    if a.order ~= b.order then return a.order < b.order end
                    return a.name < b.name
                end)
                local out = {}
                for i = 1, #parts do out[i] = parts[i].text end
                return table.concat(out, " "):gsub("%s+", " "):gsub("^%s*(.-)%s*$","%1")
            end)
            if okLbl and lbl and lbl ~= "" then label = lbl end
            return { cur=cur, max=max, txt=progTxt, label=label, cf=getObjLocation(label) }
        end
    end
    return nil
end

local equippedRodName  = "⏳ Belum ada rod"
local equippedBaitName = "⏳ Belum ada bait"

local function getPlayerStateTable(gc)
    local best, bestScore = nil, -1
    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local score = 0
            local okInv, inv = pcall(rawget, v, "Inventory")
            if okInv and type(inv) == "table" then score = score + 1 end
            local okCoins, coins = pcall(rawget, v, "Coins")
            if okCoins and type(coins) == "number" then score = score + 1 end
            local okB, baitId = pcall(rawget, v, "EquippedBaitId")
            if okB and type(baitId) == "number" then score = score + 2 end
            local okE, eqId = pcall(rawget, v, "EquippedId")
            if okE and type(eqId) == "string" and eqId ~= "" then score = score + 3 end
            local okT, eqType = pcall(rawget, v, "EquippedType")
            if okT and eqType == "Fishing Rods" then score = score + 2 end
            if score > bestScore then
                bestScore = score
                best = v
            end
        end
    end
    return best
end

-- Baca nama rod yang SEDANG DIPAKAI langsung dari memori (GC)
local function getEquippedRodName()
    local gc = getgc(true)

    local state = getPlayerStateTable(gc)
    local equippedUUID = state and rawget(state, "EquippedId") or nil

    if type(equippedUUID) ~= "string" or equippedUUID == "" then return nil end

    for i = 1, #gc do
        local v = gc[i]
        if type(v) == "table" then
            local okU, uuid = pcall(rawget, v, "UUID")
            if okU and uuid == equippedUUID then
                local okI, id = pcall(rawget, v, "Id")
                if okI and type(id) == "number" and rodTiers and rodTiers[id] then
                    local data = rodTiers[id]
                    if data and data.name then return data.name end
                    return "ID: " .. id
                end
            end
        end
    end
    return nil
end

local function equipRodByName(targetRodName)
    if not re_rod then return false end
    isEquipping = true
    addLog("🔧 [EQUIP] equipRodByName: " .. tostring(targetRodName), "debug")

    local ok, result = pcall(function()
        local gcFresh = getgc(true)
        local seenUUID = {}
        local found = nil

        for i = 1, #gcFresh do
            local v = gcFresh[i]
            if type(v) == "table" then
                local okU, uuid = pcall(rawget, v, "UUID")
                local okI, id   = pcall(rawget, v, "Id")
                if okU and okI
                    and type(uuid)=="string" and uuid:match(UUID_PAT)
                    and type(id)=="number"
                    and not seenUUID[uuid]
                then
                    seenUUID[uuid] = true
                    local data = rodTiers and rodTiers[id]
                    if data and data.name == targetRodName then
                        found = { id=id, uuid=uuid, name=data.name }
                        break
                    end
                end
            end
        end

        if not found then
            local warnKey = "equipRodByName:" .. tostring(targetRodName)
            if not _warnedMissingRod[warnKey] then
                _warnedMissingRod[warnKey] = true
                addLog("⚠ [EQUIP] Rod tidak ditemukan di GC: " .. tostring(targetRodName), "warn")
            end
            refreshRow3()
            return false
        end

        re_rod:FireServer(found.uuid, "Fishing Rods")
        task.wait(1)
        remote_equip:InvokeServer(1)

        for attempt = 1, 3 do
            task.wait(1)
            local actual = getEquippedRodName()
            if actual == found.name then
                equippedRodName = actual
                addLog(string.format("✅ [EQUIP] Verif equip rod '%s' BERHASIL (attempt %d)", actual, attempt), "ok")
                refreshRow3()
                notify("🎣 Equip rod: " .. found.name, Color3.fromRGB(100,255,180))
                return true
            end
            addLog(string.format("⚠ [EQUIP] Verif attempt %d GAGAL: Memori '%s' (butuh '%s'), retry...", attempt, tostring(actual), found.name), "warn")
            re_rod:FireServer(found.uuid, "Fishing Rods")
            task.wait(0.5)
            remote_equip:InvokeServer(1)
        end

        refreshRow3()
        return (getEquippedRodName() == found.name)
    end)

    if not ok then
        addLog("❌ [EQUIP] ERROR di equipRodByName: " .. tostring(result), "err")
    end

    isEquipping = false
    return ok and result or false
end

local function doEquipBest()
    if not re_rod or not re_bait then return end
    isEquipping = true
    addLog("🔧 [EQUIP] doEquipBest dimulai", "debug")

    local ok, err = pcall(function()
        local gcFresh = getgc(true)

        -- ── ROD ──────────────────────────────────────────────────────
        local ownedRods  = {}
        local seenUUID   = {}
        for i = 1, #gcFresh do
            local v = gcFresh[i]
            if type(v) == "table" then
                local okU, uuid = pcall(rawget, v, "UUID")
                local okI, id   = pcall(rawget, v, "Id")
                if okU and okI
                    and type(uuid)=="string" and uuid:match(UUID_PAT)
                    and type(id)=="number"
                    and not seenUUID[uuid]
                then
                    if rodTiers and rodTiers[id] then
                        seenUUID[uuid] = true
                        ownedRods[#ownedRods+1] = { id=id, uuid=uuid }
                    end
                end
            end
        end

        if rodDB and rodDB["Ghostfinn Rod"] and not _warnedMissingRod["Ghostfinn Rod"] then
            local ghostId = rodDB["Ghostfinn Rod"]
            local hasGhost = false
            for i = 1, #ownedRods do
                if ownedRods[i].id == ghostId then hasGhost = true break end
            end
            if not hasGhost then
                _warnedMissingRod["Ghostfinn Rod"] = true
                addLog("⚠ Ghostfinn Rod ada di katalog, tapi tidak ada UUID owned di GC. Tidak bisa equip sebelum benar-benar dimiliki.", "warn")
            end
        end

        local skipRodEquip = false
        local bestRod     = nil
        local bestRodTier = -1

        if activeQuestName == "Element Quest" then
            addLog("🔍 [EQUIP] Element Quest aktif: force rod '" .. ELEMENT_QUEST_FORCE_ROD .. "'", "debug")
            for _, rod in ipairs(ownedRods) do
                local data = rodTiers[rod.id]
                if data and data.name == ELEMENT_QUEST_FORCE_ROD then
                    bestRod = { id=rod.id, uuid=rod.uuid, name=data.name }
                    break
                end
            end
            if not bestRod then
                if not _warnedMissingRod["ElementQuestForceRod"] then
                    _warnedMissingRod["ElementQuestForceRod"] = true
                    addLog("⚠ Element Quest membutuhkan '" .. ELEMENT_QUEST_FORCE_ROD .. "' tapi belum dimiliki. Skip equip rod lain.", "warn")
                end
                skipRodEquip = true
            end
        else
            addLog(string.format("🔍 [EQUIP] Mencari rod terbaik (total di GC: %d)", #ownedRods), "debug")
            for _, rod in ipairs(ownedRods) do
                local data = rodTiers[rod.id]
                if data then
                    if not bestRod then
                        bestRodTier = data.tier
                        bestRod = { id=rod.id, uuid=rod.uuid, name=data.name }
                    else
                        local candPref = _equipRodRankByName[data.name]
                        local bestPref = _equipRodRankByName[bestRod.name]
                        if candPref and bestPref then
                            if candPref < bestPref or (candPref == bestPref and rod.id < bestRod.id) then
                                bestRodTier = data.tier
                                bestRod = { id=rod.id, uuid=rod.uuid, name=data.name }
                            end
                        else
                            if data.tier > bestRodTier then
                                bestRodTier = data.tier
                                bestRod = { id=rod.id, uuid=rod.uuid, name=data.name }
                            elseif data.tier == bestRodTier then
                                local candRank = _rodRankByName[data.name] or math.huge
                                local bestRank = _rodRankByName[bestRod.name] or math.huge
                                if candRank < bestRank or (candRank == bestRank and rod.id < bestRod.id) then
                                    bestRod = { id=rod.id, uuid=rod.uuid, name=data.name }
                                end
                            end
                        end
                    end
                end
            end
        end

        if skipRodEquip then
            refreshRow3()
        elseif bestRod then
            addLog(string.format("🛒 [EQUIP] Mencoba equip rod: '%s' (uuid: %s)", bestRod.name, bestRod.uuid:sub(1,8).."..."), "info")
            re_rod:FireServer(bestRod.uuid, "Fishing Rods")

local function getNextBuyTarget(cachedRods, cachedBaits)
    local withPrice = {}
    local nilPrice  = {}
    for _, rodName in ipairs(TARGET_RODS) do
        local rodId = rodDB[rodName]
        if rodId and not isRodOwned(rodId, cachedRods) then
            if isBuyBlocked("rod", rodId) then
                continue
            end
            local price = rodPriceDB[rodId] or 0
            if price > 0 then withPrice[#withPrice+1] = { name=rodName, id=rodId, kind="rod", price=price }
            else nilPrice[#nilPrice+1] = rodName end
        end
    end
    for _, baitEntry in ipairs(TARGET_BAITS) do
        local baitId = baitDB[baitEntry.name]
        if baitId and not isBaitOwned(baitId, cachedBaits) then
            if isBuyBlocked("bait", baitId) then
                continue
            end
            local price = baitPriceDB[baitId] or 0
            if price > 0 then withPrice[#withPrice+1] = { name=baitEntry.name, id=baitId, kind="bait", price=price }
            else nilPrice[#nilPrice+1] = baitEntry.name end
        end
    end
    table.sort(withPrice, function(a, b) return a.price < b.price end)
    if #withPrice > 0 then
        local c = withPrice[1]
        local pct = math.min(100, math.floor(currentCoins / c.price * 100))
        local suffix = #nilPrice > 0 and string.format("  (+%d ⏳NIL)", #nilPrice) or ""
        if c.kind == "rod" then
            return string.format("Next: %s (id=%d, %s) [%d%%]%s", c.name, c.id, formatCoins(c.price), pct, suffix)
        else
            return string.format("Next: %s (id=%d, %s) [%d%%]%s", c.name, c.id, formatCoins(c.price), pct, suffix)
        end
    elseif #nilPrice > 0 then
        return "⏳ Menunggu harga: " .. table.concat(nilPrice, ", ")
    end
    return "✓ Semua item sudah dibeli"
end

-- ============================================================
--  waitForCatch — FIX race condition
-- ============================================================
local function waitForCatch()
    local caught = false
    local done   = false

    local conn
    conn = catchEvent.Event:Connect(function()
        if not done then
            caught = true
            done   = true
        end
    end)

    if fishLatch then
        caught    = true
        done      = true
        fishLatch = false
    end

    if not done then
        task.delay(CATCH_TIMEOUT, function()
            if not done then done = true end
        end)
        while not done do task.wait() end
    end

    conn:Disconnect()
    fishLatch = false

    return caught
end

local pendingEquip = {}  -- { id=number, kind="rod"/"bait", name=string }

local function tryEquipPending()
    if #pendingEquip == 0 then return end
    isEquipping = true
    addLog("🔧 [PENDING] tryEquipPending dimulai", "debug")
    
    local ok, err = pcall(function()
        local gcFresh = getgc(true)
        local stillPending = {}
        local anyFound = false
        for _, item in ipairs(pendingEquip) do
            local found = false

            if item.kind == "rod" then
                -- cari UUID rod spesifik di GC
                local seenUUID = {}
                for i = 1, #gcFresh do
                    local v = gcFresh[i]
                    if type(v) == "table" then
                        local okU, uuid = pcall(rawget, v, "UUID")
                        local okI, id   = pcall(rawget, v, "Id")
                        if okU and okI
                            and type(uuid)=="string" and uuid:match(UUID_PAT)
                            and type(id)== "number" and id == item.id
                            and not seenUUID[uuid]
                        then
                            if rodTiers and rodTiers[id] then
                                seenUUID[uuid] = true
                                addLog("🔄 [PENDING] Rod: " .. item.name .. " (uuid: " .. uuid:sub(1,8) .. ") ditemukan di GC, equip!", "ok")
                                re_rod:FireServer(uuid, "Fishing Rods")
                                task.wait(0.5)
                                remote_equip:InvokeServer(1)

                                -- Verifikasi singkat
                                task.wait(1)
                                local currentRod = getEquippedRodName()
                                if currentRod == nil or currentRod == "" then
                                    addLog("⚠ [PENDING] Verif rod: state belum siap (EquippedId nil), akan di-retry doEquipBest", "warn")
                                elseif currentRod == item.name then
                                    addLog("✅ [PENDING] Verif equip rod '" .. item.name .. "' BERHASIL", "ok")
                                else
                                    addLog("⚠ [PENDING] Verif rod '" .. item.name .. "' GAGAL, akan di-retry doEquipBest", "warn")
                                end

                                equippedRodName = item.name
                                refreshRow3()
                                found = true
                                anyFound = true
                                break
                            end
                        end
                    end
                end

            elseif item.kind == "bait" then
                -- cari bait di inventory GC
                for i = 1, #gcFresh do
                    local v = gcFresh[i]
                    if type(v) == "table" then
                        local okE, eqBait = pcall(rawget, v, "EquippedBaitId")
                        local okI, inv    = pcall(rawget, v, "Inventory")
                        if okE and okI and type(eqBait)=="number" and type(inv)=="table" then
                            local okB, baits = pcall(rawget, inv, "Baits")
                            if okB and type(baits)=="table" then
                                pcall(function()
                                    for _, bait in pairs(baits) do
                                        if type(bait)=="table" then
                                            local okId, id = pcall(rawget, bait, "Id")
                                            if okId and id == item.id then
                                                addLog("🔄 [PENDING] Bait: " .. item.name .. " ditemukan di GC, equip!", "ok")
                                                re_bait:FireServer(item.id)
                                                
                                                -- Verifikasi singkat
                                                task.wait(1)
                                                local currentBait = getEquippedBaitName()
                                                if currentBait == item.name then
                                                    addLog("✅ [PENDING] Verif equip bait '" .. item.name .. "' BERHASIL", "ok")
                                                else
                                                    addLog("⚠ [PENDING] Verif bait '" .. item.name .. "' GAGAL, akan di-retry doEquipBest", "warn")
                                                end
                                                
                                                equippedBaitName = item.name
                                                refreshRow3()
                                                found = true
                                                anyFound = true
                                            end
                                        end
                                    end
                                end)
                                break
                            end
                        end
                    end
                end
            end

            if not found then
                stillPending[#stillPending+1] = item
                addLog(string.format("⏳ [PENDING] %s id=%d belum muncul di GC, tetap di antrian", item.name, item.id), "debug")
            end
        end
        pendingEquip = stillPending
        if anyFound and AUTO_EQUIP_BEST then
            task.wait(0.5)
            doEquipBest()
        end
    end)

    if not ok then
        addLog("❌ [PENDING] ERROR di tryEquipPending: " .. tostring(err), "err")
    end
    isEquipping = false
end

    local caught = waitForCatch()

    if caught then
        local cycleMs = (tick() - castTime) * 1000
        if avgCycleMs == 0 then
            avgCycleMs = cycleMs
        else
            avgCycleMs = avgCycleMs * 0.8 + cycleMs * 0.2
        end
    end

    if not calibrated then
        -- ── FASE KALIBRASI ──────────────────────────────────────────
        if caught then
            if calibConfirm == 0 then
                -- Catch pertama: catat delay ini sebagai base konfirmasi
                calibCatchDelay = REEL_DELAY
                addLog(string.format("🎯 Catch pertama di delay %.1f — mulai konfirmasi 5x", calibCatchDelay), "info")
            end
            calibConfirm = calibConfirm + 1
            row1Val.Text = string.format(
                "⚙ Konfirmasi %d/5  P:%.1f R:%.1f",
                calibConfirm, ARG_POWER, REEL_DELAY)

            if calibConfirm >= 5 then
                calibrated      = true
                calibConfirm    = 0
                missCounter     = 0
                optimalDelay    = REEL_DELAY
                fpm_count       = 0
                fpm_start       = tick()
                totalCaught     = 0
                totalMiss       = 0
                avgCycleMs      = 0
                notify(string.format(
                    "✓ Kalibrasi selesai! 5/5 berhasil. Delay: %.1f", optimalDelay),
                    Color3.fromRGB(0,220,80))
            end
        else
            if calibConfirm > 0 then
                -- Konfirmasi gagal: kembali ke delay yang TERBUKTI catch (bukan optimalDelay lama)
                notify(string.format(
                    "✗ Konfirmasi gagal di %d/5. Kembali ke %.1f",
                    calibConfirm, calibCatchDelay),
                    Color3.fromRGB(255,100,60))
                addLog(string.format("⚠ Konfirmasi gagal %d/5. Reset ke calibCatchDelay=%.1f (bukan optimalDelay=%.1f)",
                    calibConfirm, calibCatchDelay, optimalDelay), "warn")
                calibConfirm = 0
                ARG_POWER    = calibCatchDelay
                REEL_DELAY   = calibCatchDelay
                avgCycleMs   = 0
            else
                ARG_POWER  = math.min(MAX_VAL, ARG_POWER  + STEP)
                REEL_DELAY = math.min(MAX_VAL, REEL_DELAY + STEP)
            end
            row1Val.Text = string.format(
                "⚙ Kalibrasi  P:%.1f R:%.1f", ARG_POWER, REEL_DELAY)
        end
    else
        -- ── FASE NORMAL (sudah terkalibrasi) ────────────────────────
        if caught then
            totalCaught = totalCaught + 1
            fpm_count   = fpm_count + 1
            -- missCounter TIDAK direset saat catch (akumulasi total)
        else
            totalMiss   = totalMiss + 1
            missCounter = missCounter + 1

            -- Jika total miss sejak kalibrasi terakhir ≥ threshold → recalibrate
            if missCounter >= MISS_THRESHOLD then
                notify(string.format(
                    "⚠ %d miss total sejak kalibrasi! Recalibrate dari optimal (%.1f)",
                    missCounter, optimalDelay),
                    Color3.fromRGB(255, 160, 0))
                calibrated    = false
                calibConfirm  = 0
                missCounter   = 0
                avgCycleMs    = 0
                -- Mulai dari optimalDelay, bukan 0.1
                ARG_POWER     = optimalDelay
                REEL_DELAY    = optimalDelay
                calibCatchDelay = optimalDelay   -- reset base konfirmasi ke optimal terakhir
                row1Val.Text  = string.format("⚙ Recal dari %.1f...", optimalDelay)
            end
        end

        local fpm = fpm_count / math.max((tick() - fpm_start) / 60, 0.01)
        row1Val.Text = formatRow1(totalCaught, totalMiss, fpm, avgCycleMs)
    end

    task.wait(0.3)
end
